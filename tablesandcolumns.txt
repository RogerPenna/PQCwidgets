  def html_resumonotas(rec, table):
    # Fetch all sessions
    sessoes = list(Sessoes.all)  # Convert the RecordSet to a list

    # Fetch PDCA records for the given company
    pdca_records = PDCA.lookupRecords(EmpresaRef=rec.id)

    # Define colors and styles
    session_colors = ["#1F3864"]  # Blue tonality
    session_text_color = "#FFFFFF"  # White
    column_title_background = "#D8D8D8"  # Gray
    font_family = "Arial"

    # Define the subsessions
    subsessions = [
        "1- Pensamento Sistêmico",
        "2- Compromisso com as Partes Interessadas",
        "3 - Aprendizado Organizacional e Inovação",
        "4 - Adaptabilidade",
        "5 - Liderança Transformadora",
        "6 - Desenvolvimento Sustentável",
        "7 - Orientação por Processos",
        "8 - Resultados"
    ]

    # Helper functions
    def get_unique_practice_scores(practice_ids, score_field):
        unique_practice_ids = list(set(practice_ids))
        scores = [getattr(record, score_field) for record in pdca_records if record.PraticaRef in unique_practice_ids]
        return scores

    def count_practice_occurrences(practices):
        occurrences = {}
        for practice in practices:
            if practice not in occurrences:
                occurrences[practice] = 0
            occurrences[practice] += 1
        unique_practices_count = len(occurrences)
        return occurrences, unique_practices_count

    def calculate_session_scores(score_field):
        session_scores = []
        for i, sessao in enumerate(sessoes):
            subsessoes = Subsessoes.lookupRecords(RefSessao=sessao.id)
            sessao_pontuacao_maxima = 0
            sessao_atingida = 0

            for subsessao in subsessoes:
                praticas = Questoes.lookupRecords(RefSub=subsessao.id)
                practice_counts, unique_practices_count = count_practice_occurrences([p for pratica in praticas for p in pratica.Praticas])
                subsessao_pontuacao_maxima = subsessao.PontuacaoMaxima
                subsessao_atingida = 0

                for pratica in praticas:
                    pratica_ids = list(pratica.Praticas)  # Convert RecordSet to list
                    scores = get_unique_practice_scores(pratica_ids, score_field)
                    average_score = round(AVERAGE(scores), 2)
                    practice_count_in_row = practice_counts[pratica_ids[0]]
                    percentual = round(average_score / practice_count_in_row, 2)
                    pontuacao = round(subsessao.PontuacaoMaxima / (practice_counts[pratica_ids[0]] * unique_practices_count), 2)
                    alcancada = round((percentual * pontuacao / 100) / practice_count_in_row, 2)
                    subsessao_atingida += alcancada

                sessao_pontuacao_maxima += subsessao.PontuacaoMaxima
                sessao_atingida += subsessao_atingida

            sessao_percentage = round((sessao_atingida / sessao_pontuacao_maxima) * 100, 2) if sessao_pontuacao_maxima > 0 else 0
            session_scores.append(sessao_percentage)
        return session_scores

    def get_color_gradient(percentage):
        # Calculate the gradient color from red (0%) to yellow (50%) to green (100%)
        if percentage <= 50:
            red = 255
            green = int((percentage / 50) * 255)
        else:
            green = 255
            red = int(((100 - percentage) / 50) * 255)
        blue = 0
        return f"rgb({red},{green},{blue})"

    # Calculate scores for Autoavaliação and Avaliação Externa
    auto_avaliacao_scores = calculate_session_scores('Nota_da_Pratica')
    avaliacao_externa_scores = calculate_session_scores('NotaPratExt')

    # Generate the HTML report
    subsession_html = []
    for i, subsession_name in enumerate(subsessions):
        if i < len(auto_avaliacao_scores) and i < len(avaliacao_externa_scores):
            auto_avaliacao = auto_avaliacao_scores[i]
            avaliacao_externa = avaliacao_externa_scores[i]
            auto_avaliacao_color = get_color_gradient(auto_avaliacao)
            avaliacao_externa_color = get_color_gradient(avaliacao_externa)
            subsession_html.append("""
            <tr>
                <td style="border: 1px solid black; padding: 5px; background-color: {0}; color: white;">{1}</td>
                <td style="border: 1px solid black; padding: 5px; background-color: {2}; text-align: center; color: white;">{3:.2f}%</td>
                <td style="border: 1px solid black; padding: 5px; background-color: {4}; text-align: center; color: white;">{5:.2f}%</td>
            </tr>
        """.format(
                session_colors[0],
                subsession_name,
                auto_avaliacao_color,
                auto_avaliacao,
                avaliacao_externa_color,
                avaliacao_externa
            ))

    # Generate the final HTML report
    html_report = """
<html>
  <head>
    <title>Relatório de Avaliações</title>
  </head>
  <body>
    <h1 style="text-align: center; font-family: {1}; font-size: 12px; color: white;">Relatório de Avaliações</h1>
    <table style="width: 100%; border-collapse: collapse; font-family: {1}; font-size: 10px;">
        <thead>
            <tr style="background-color: {0}; color: white;">
                <th style="border: 1px solid black; padding: 5px;">Subsessões</th>
                <th style="border: 1px solid black; padding: 5px;">Autoavaliação</th>
                <th style="border: 1px solid black; padding: 5px;">Avaliação Externa</th>
            </tr>
        </thead>
        <tbody>
            {2}
        </tbody>
    </table>
  </body>
</html>
""".format(
        session_colors[0],
        font_family,
        '\n'.join(subsession_html)
    )

    return html_report

  def Formula_Resultados(rec, table):
    def gerar_relatorio_html():
        # Define colors for indicator types
        cores_tipos = {
            "8.1 Econômico-financeiros": "#084794",
            "8.3 Relativos a clientes e mercados": "#460D81",
            "8.2 Sociais e ambientais": "#760C6E",
            "8.4 Relativos às pessoas": "#0C686A",
            "8.5 Relativos aos processos": "#126E0E"
        }

        # Define reference values for indicator types
        valores_referencia = {
            "8.1 Econômico-financeiros": 420,
            "8.3 Relativos a clientes e mercados": 180,
            "8.2 Sociais e ambientais": 100,
            "8.4 Relativos às pessoas": 210,
            "8.5 Relativos aos processos": 490
        }
    
        # Get all `Resultados` records for this company (using `$id` to refer to the current company's ID)
        resultados = Resultados.lookupRecords(RefEmpresa=rec.id)
    
        # Extract unique indicator types for this company
        tipos_indicadores = set(resultado.Num_TipoIndicRef for resultado in resultados)
    
        # Generate HTML report
        html_parts = []

        # Iterate over all unique indicator types
        for tipo in tipos_indicadores:
            # Map the type to its full name
            tipo_nome = next((nome_completo for nome_completo in cores_tipos.keys() if nome_completo.startswith(tipo)), None)
        
            if tipo_nome is None:
                continue  # Skip if the type is not found in the dictionary
        
            # Filter results for the current indicator type
            resultados_filtrados = [resultado for resultado in resultados if resultado.Num_TipoIndicRef == tipo]
        
            # Sum the scores of the filtered results
            soma_pontuacao = sum(resultado.PONTUACAO for resultado in resultados_filtrados)
        
            # Get the background color corresponding to the indicator type
            cor_fundo = cores_tipos[tipo_nome]

            # Calculate the percentage
            valor_referencia = valores_referencia[tipo_nome]
            porcentagem = (soma_pontuacao / valor_referencia) * 100 if valor_referencia else 0
        
            # Add the indicator type and score to the HTML with formatting
            html_parts.append(f"""
        <div style="background-color: {cor_fundo}; color: white; padding: 8px; margin-bottom: 4px;">
            {tipo_nome}
        </div>
        <div style="display: flex; padding: 8px; margin-bottom: 12px;">
            <div style="flex: 1;">{soma_pontuacao:.2f}</div>
            <div style="flex: 1; text-align: right;">{porcentagem:.2f}%</div>
        </div>
        """)

        # Join all parts of the HTML
        relatorio_html = ''.join(html_parts)
        return relatorio_html

    # Call the function to generate the HTML report for this company
    return gerar_relatorio_html()

    def verifica_melhoria(c2022, c2023, c2024, direcao, num_ref):
        # Lista de indicadores que devem apresentar melhoria se forem 100% em todos os anos
        indicadores_especiais = {7, 12, 16, 19, 20, 21, 22}

        # Verifica se num_ref é uma referência e tenta extrair o número real
        num = num_ref if isinstance(num_ref, (int, float)) else getattr(num_ref, 'Num', None)

        # Verifica se há dados nos três anos e se não são zero
        if c2022 is None or c2023 is None or c2024 is None or c2022 == 0 or c2023 == 0 or c2024 == 0:
            return "N/A"

        # Se o indicador está na lista especial e todas as colunas são aproximadamente 100%, retorna "SIM"
        if num in indicadores_especiais and abs(c2022 - 100) < 0.01 and abs(c2023 - 100) < 0.01 and abs(c2024 - 100) < 0.01:
            return "SIM"

        # Verifica a direção do indicador
        if direcao == "⬆":
            # Determina a direção da melhoria (maior é melhor)
            if c2022 < c2023 < c2024:
                return "SIM"
            else:
                return "NÃO"
        elif direcao == "⬇":
            # Determina a direção da melhoria (menor é melhor)
            if c2022 > c2023 > c2024:
                return "SIM"
            else:
                return "NÃO"
        else:
            return "N/A"

        unique_practice_ids = list(set(practice_ids))
        scores = [getattr(record, score_field) for record in pdca_records if record.PraticaRef in unique_practice_ids]
        return scores

    def calculate_average_percentual(practices, score_field):
        total_percentual = 0
        count = 0
        for practice_id in practices:
            scores = get_unique_practice_scores([practice_id], score_field)
            if scores:
                total_percentual += sum(scores) / len(scores)
                count += 1
        return round(total_percentual / count, 2) if count > 0 else 0

    def calculate_session_score(sessao, score_field):
        subsessoes = Subsessoes.lookupRecords(RefSessao=sessao.id)
        sessao_pontuacao_maxima = 0
        sessao_atingida = 0

        for subsessao in subsessoes:
            praticas = Questoes.lookupRecords(RefSub=subsessao.id)
            all_practices = [p for pratica in praticas for p in pratica.Praticas]
            unique_practices = list(set(all_practices))
            average_subsessao_percentual = calculate_average_percentual(unique_practices, score_field)
            subsessao_pontuacao_maxima = subsessao.PontuacaoMaxima
            alcancada = round((average_subsessao_percentual * subsessao_pontuacao_maxima) / 100, 2)
        
            sessao_pontuacao_maxima += subsessao_pontuacao_maxima
            sessao_atingida += alcancada

        return sessao_percentage
